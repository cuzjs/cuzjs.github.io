Redux

下面是一个标准 Action + Reducer 的写法, 供参考:

import { createReducer } from 'cuz';

const FETCH_DATA = 'fetch data';

export function fetchData() {
  return {
    type: FETCH_DATA,
    promise: (request) => request.get(`/data`)
  };
}

const initState = {
  data: [],
}

export default createReducer(initState, {
  [FETCH_DATA](state, action) {
    return {
      ...state,
      data: action.result.retValue,
    }
  }
});
