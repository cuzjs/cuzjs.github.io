Redux

下面是一个 Redux 的 action + reducer 的写法:

// constants/book.js

export const FETCH_BOOKS = '获取图书列表';

// actions/book.js

import { FETCH_BOOKS } from '../constants/book';

export function fetchBooks() {
  return {
    type: FETCH_DATA,
    promise: (request) => request.get(`/data`)
  };
}

// reducers/book.js

import { createReducer } from 'cuz';
import { FETCH_BOOKS } from '../constants/book';

const initState = {
  data: [],
}

export default createReducer(initState, {
  [FETCH_BOOKS](state, action) {
    return {
      ...state,
      data: action.result.retValue,
    }
  }
});
